# cURL Commands for All APIs
# Server: http://localhost:4567
# Note: Replace placeholders like {id}, {token}, etc. with actual values.
# For authenticated endpoints, include Authorization header if needed.

## Authentication
# Login to get JWT token
curl -X POST http://localhost:4567/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{"username": "your_username", "password": "your_password"}'

## Admin URL Requests
# Get all URL requests
curl -X GET http://localhost:4567/api/admin/url-requests \
  -H "Authorization: Bearer {token}"

# Get URL request by ID
curl -X GET http://localhost:4567/api/admin/url-requests/{id} \
  -H "Authorization: Bearer {token}"

# Create URL request
curl -X POST http://localhost:4567/api/admin/url-requests \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer {token}" \
  -d '{
    "baseUrl": "https://api.example.com",
    "tile": "DASHBOARD_POLICIES",
    "description": "Policy management dashboard",
    "application": {
      "id": 1
    },
    "environment": {
      "id": 1
    },
    "section": {
      "id": 1
    }
  }'

# Update URL request
curl -X PUT http://localhost:4567/api/admin/url-requests/{id} \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer {token}" \
  -d '{
    "baseUrl": "https://updated.com",
    "tile": "tile1",
    "description": "Updated URL",
    "application": {"id": 1},
    "environment": {"id": 1},
    "section": {"id": 1}
  }'

# Delete URL request
curl -X DELETE http://localhost:4567/api/admin/url-requests/{id} \
  -H "Authorization: Bearer {token}"

## URLs Parallel
# Fetch a list of URLs
curl -X GET http://localhost:4567/api/urls-parallel/fetch

# Call all URLs in parallel and return their statuses
curl -X POST http://localhost:4567/api/urls-parallel/call \
  -H "Content-Type: application/json" \
  -d '{
    "urls": [
      "https://www.google.com",
      "https://www.github.com"
    ]
  }'
